Solve {
	NewCurrentFile="n@node@_ini"
	Set (traps(Value = 0 Frozen))

    	# initial solution
	Coupled (Iterations=100) { Poisson }
    Coupled (Iterations=100) { Poisson Electron Hole }

	Set (Traps(-Frozen) )

	NewCurrentFile="n@node@_pgm_"
	Transient(
	    InitialTime=0. FinalTime=!(puts [expr $t1p+2.5e-7])! Increment= 1.4 Decrement= 2.
	    InitialStep= 1e-12 Minstep= 1.e-15 MaxStep= 1e-3
	){
	    Coupled { Poisson Electron Hole }
	    !(
		set dt 1.50e-4
	    set tR 2.00e-4
	    set t1p 5e-7
		set Vg @Vgpi@
	    set delVg [expr @<(Vgpf-Vgpi)/Steps>@]
	    for { set i 1 } { $i <= @Steps@ } { incr i } {
	    	set t2p [expr $t1p + 1.5e-7]
	        set t3p [expr $t1p + $dt]
	        set t4p [expr $t2p + $dt]

	        set t1p [expr $t4p + $tR]
				puts "Plot(FilePrefix=\"n@node@_pgmr_$Vg\" Time=([format %0.6e [expr $t1p]]))"
				puts "Save(FilePrefix=\"n@node@_pgmr_$Vg\" Time=([format %0.6e [expr $t1p]]))"
			set Vg  [expr $Vg + $delVg]

	    }
        )!
	}
}
